// Dialog Functionality
function openDialog() {
    const dialog = document.getElementById('test-dialog');
    const backdrop = document.getElementById('dialog-backdrop');
    
    if (dialog && backdrop) {
        dialog.classList.add('dialog-open');
        backdrop.classList.add('dialog-open');
        document.body.style.overflow = 'hidden';
    }
}
function closeDialog() {
    const dialog = document.getElementById('test-dialog');
    const backdrop = document.getElementById('dialog-backdrop');
    
    if (dialog && backdrop) {
        dialog.classList.remove('dialog-open');
        backdrop.classList.remove('dialog-open');
        document.body.style.overflow = '';
    }
}
// Event Listeners for Dialog
document.addEventListener('DOMContentLoaded', function() {
    const openDialogBtn = document.getElementById('open-dialog-btn');
    const closeDialogBtn = document.getElementById('close-dialog-btn');
    const dialogBackdrop = document.getElementById('dialog-backdrop');
    
    // Open dialog
    if (openDialogBtn) {
        openDialogBtn.addEventListener('click', function() {
            console.log("Open dialog button clicked");
            openDialog();
        });
    }
    
    // Close dialog
    if (closeDialogBtn) {
        closeDialogBtn.addEventListener('click', function() {
            console.log("Dialog close button clicked");
            closeDialog();
        });
    }
    
    // Close dialog when clicking backdrop
    if (dialogBackdrop) {
        dialogBackdrop.addEventListener('click', function(e) {
            // Only close if the backdrop itself is clicked
            if (e.target === dialogBackdrop) {
                console.log("Dialog backdrop clicked");
                closeDialog();
            }
        });
    }
    
    // Close dialog with Escape key
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            closeDialog();
        }
    });
});
